function calculateTactics() { // 1. 讀取強度差距 const diff = document.getElementById('strength-diff').value; // 讀取雙方位置人數 const getVal = (side, pos) => parseInt(document.getElementById(`${side}-${pos}`).value) || 0; const my = { DL: getVal('my','DL'), DC: getVal('my','DC'), DR: getVal('my','DR'), DMC: getVal('my','DMC'), ML: getVal('my','ML'), MC: getVal('my','MC'), MR: getVal('my','MR'), AML: getVal('my','AML'), AMC: getVal('my','AMC'), AMR: getVal('my','AMR'), ST: getVal('my','ST') }; const opp = { DL: getVal('opp','DL'), DC: getVal('opp','DC'), DR: getVal('opp','DR'), DMC: getVal('opp','DMC'), ML: getVal('opp','ML'), MC: getVal('opp','MC'), MR: getVal('opp','MR'), AML: getVal('opp','AML'), AMC: getVal('opp','AMC'), AMR: getVal('opp','AMR'), ST: getVal('opp','ST') }; // 計算我方實質後衛總數 const myTotalDefenders = my.DL + my.DC + my.DR; // 2. 核心戰術心態與基礎設定 let mentality = "正常"; let counter = "關閉"; let press = "低"; let tackle = "一般"; if(diff === "very-strong") { mentality = "猛攻"; counter = "關閉"; press = "高"; tackle = "一般"; } else if(diff === "strong") { mentality = "進攻"; counter = "關閉"; press = "高"; tackle = "一般"; } else if(diff === "equal") { mentality = "正常"; counter = "關閉"; press = "低"; tackle = "一般"; } else if(diff === "weak") { mentality = "防守"; counter = "開啟"; press = "低"; tackle = "一般"; } else if(diff === "very-weak") { mentality = "死守"; counter = "開啟"; press = "低"; tackle = "保守"; } // 傳球方向優化邏輯 let oppFlankDef = opp.DL + opp.DR + opp.ML + opp.MR + opp.AML + opp.AMR; let oppMidDef = opp.DC + opp.DMC + opp.MC + opp.AMC; let myFlankAtt = my.ML + my.MR + my.AML + my.AMR; let myMidAtt = my.MC + my.AMC + my.ST; let direction = "混合"; if(oppMidDef > oppFlankDef && myFlankAtt >= 2) { direction = "兩翼"; } else if (oppFlankDef > oppMidDef && myMidAtt >= 3) { direction = "中路"; } // 【修正核心】動態調整傳球方式與防越位機制 let type = "短傳"; let offside = "關閉"; if (diff === "weak" || diff === "very-weak") { type = "長傳"; // 弱勢打防反長傳 offside = "關閉"; } else if (diff === "equal") { type = "混合"; offside = "關閉"; } else { // 我方具備優勢 (strong / very-strong) type = "短傳"; // 【防越位修正】唯有我方屬於「標準4後衛(有DL與DR)」且開進攻時，才允許開啟越位陷阱 if (myTotalDefenders >= 4 && my.DL > 0 && my.DR > 0) { offside = "開啟"; } else { offside = "關閉"; // 3後衛或殘缺防線死都不開越位 } } // 盯人方式 let marking = "區域防守"; if(opp.ST >= 2 && my.DC >= opp.ST) { marking = "人盯人防守"; } // 渲染戰術面板 document.getElementById('t-mentality').innerText = mentality; document.getElementById('t-direction').innerText = direction; document.getElementById('t-type').innerText = type; document.getElementById('t-counter').innerText = counter; document.getElementById('t-press').innerText = press; document.getElementById('t-tackle').innerText = tackle; document.getElementById('t-marking').innerText = marking; document.getElementById('t-offside').innerText = offside; // 3. 【箭頭邏輯大修正】拒絕無腦紅箭頭，講究攻防層次 const arrowListContainer = document.getElementById('arrow-output'); arrowListContainer.innerHTML = ""; const positionsList = ['DL','DC','DR','DMC','ML','MC','MR','AML','AMC','AMR','ST']; positionsList.forEach(pos => { let count = my[pos]; if(count > 0) { let arrowText = "無箭頭 (平衡)"; let arrowClass = "arrow-none"; // 邊後衛防守判定 if(pos === 'DL' || pos === 'DR') { let oppWingAtt = (pos === 'DL') ? (opp.MR + opp.AMR) : (opp.ML + opp.AML); if(oppWingAtt === 0 && (diff === "strong" || diff === "very-strong")) { arrowText = "⬆️ 紅箭頭 (插上助攻)"; arrowClass = "arrow-up"; } else if (oppWingAtt > 1 || diff === "very-weak" || diff === "weak") { arrowText = "⬇️ 藍箭頭 (留在後方防守)"; arrowClass = "arrow-down"; } } // 核心防守球員判定 else if(pos === 'DMC' || pos === 'DC') { if(diff === "weak" || diff === "very-weak" || opp.AMC > 0 || opp.ST >= 2) { arrowText = "⬇️ 藍箭頭 (深蹲防守)"; arrowClass = "arrow-down"; } } // 【攻擊手箭頭層次修正】 else if(pos === 'ST') { // 只有在我方具備實力優勢時，前鋒才掛紅箭頭衝鋒吃餅 if(diff === "strong" || diff === "very-strong") { arrowText = "⬆️ 紅箭頭 (全速突擊吃餅)"; arrowClass = "arrow-up"; } } else if(pos === 'AMC') { // 前腰（AMC）作為中路方向盤，堅決不掛紅箭頭，留在弧頂做球遠射 arrowText = "無箭頭 (負責弧頂盤帶與遠射)"; arrowClass = "arrow-none"; } else if(pos === 'AML' || pos === 'AMR') { // 邊鋒具備優勢時可以掛紅箭頭進行單兵爆破 if(diff === "strong" || diff === "very-strong") { arrowText = "⬆️ 紅箭頭 (兩翼降維打擊)"; arrowClass = "arrow-up"; } } // 【邊中場兼顧防守修正】 else if(pos === 'ML' || pos === 'MR') { // 只有當我方是標準4後衛（後方有DL/DR扛著），邊中場才能放膽掛紅箭頭突擊 if(direction === "兩翼" && (diff === "strong" || diff === "very-strong") && my.DL > 0 && my.DR > 0) { arrowText = "⬆️ 紅箭頭 (邊路突擊)"; arrowClass = "arrow-up"; } else { arrowText = "無箭頭 (維持中場平衡，兼顧防守)"; arrowClass = "arrow-none"; } } else if(pos === 'MC') { // 中前衛除非極端情況，保持平衡控球 arrowText = "無箭頭 (掌控球權中轉)"; arrowClass = "arrow-none"; } // 將成果加到畫面上 const card = document.createElement('div'); card.className = `arrow-card ${arrowClass}`; card.innerHTML = `
${pos}
${arrowText}
`; arrowListContainer.appendChild(card); } }); }
